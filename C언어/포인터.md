포인터

```c
int *a; // 포인터 변수 선언, a라는 변수 공간을 새로 생성
*a = 10; // *a -> a의 값, *a=10 -> a의 값(번지)에다가 10을 넣어라
// 만약 a가 102번지를 가지고 있으면 102번지에 10을 넣음, 102번지의 값은 10
printf("%d", a); // a의 위치, 102
printf("%d", *a); // a의 값 = a의 위치의 값 = 102번지의 값  10

//////////////////////

int b = 10; // 그냥 변수
int *a; // 포인터변수 선언
a = &b; // &b = b의 위치
printf("%d\n", *a);
printf("%d\n", b);
```

배열과 포인터의 관계

```c
int a[5]; // 5개의 길이를 가진 배열 a 선언
int *b = a; // 포인터 변수 b의 값에 배열 a를 넣음
a[2] = 10;  // 배열 a의 세번째 공간에 10을 넣음
b[2] = 20; // 포인터 변수 b의 세번째 공간에 20의 값을 넣음, b의 3번째 공간은 a의 3번째 공간
// a[2] = 20이 된다.
```

2차원 배열과 포인터

```c
int a[3][2] = {{1,2},{3,4},{5,6}};
/*
1(100) | 2(101) = 대표행은 첫번째 주소값을 가진다. 100
3(102) | 4(103) = 102
5(104) | 6(105) = 104
*/
// a[0]은 100을 가리킨다
a = 100번지; 
*a = 100번지;
**a  = 1 //a의 값의 값

/////////////

int darr[3][3] = {{1,2,3},{4,5,6},{7,8,9}};
// *darr은 darr[0]의 번지를 가리킨다. darr의 값은 100
// **darr은 *darr[0]이다, darr[0]의 값은 1 
int sum1, sum2;
sum1 = *(*darr+1) + *(*darr+2); // *darr = 100
sum2 = *darr[1] + *darr[2];
printf("%d,%d",sum1,sum2);
```
