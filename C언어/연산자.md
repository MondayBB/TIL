```c
// 조건식 안에 조건이 증감연산자로 계산됐으면 그것도 계산한걸로 쳐준다
int i = 3;
int j = 4;
if((++i > j--)&&(i++ < --j))i = i-- ++j;
else j = i-- - --j;
// i = 3, j = 2
```

```c
//출력형식
%d = 10진수로 출력, 데이터 타입 - 정수, 문자
%f = 고정 소수점으로 출력, 데이터 타입 - 정수, 실수
%c = 문자 하나 출력
%s = 문자열
%x = 16진수
%o = 8진수
```

```c
("%5d", 13); // vvv13
("%05d", 13); // 00013

("%f",3.26); // 3.260000
("%6.1f", 3.3); // vvv3.3, 6개 공간을 만들고 소수점 첫째자리까지 표현, .도 공간에 포함
("%-6.1f", 3.26); // 3.3vvv, 소수점 첫째자리까지 표현하니까 둘째에서 반올림
("%-6.3f", 3.26); // 3.260v
```

비트연산자

```c
&  두개가 참일 때만 참이다 
| 둘중 하나만 참이어도 참인것
^ 두개가 틀려야 참인 것
~ 값 +1 하고 양수면 음수, 음수면 양수로 변환
>> 우측으로 밀기 10>>2 // 1010 -> 0010
<< 좌측으로 밀기 10<<2 // 1010 -> 101000
```
