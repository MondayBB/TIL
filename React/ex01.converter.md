### Minutes & Hours
```jsx
function MinutesToHours(){
  // input 값을 받아오는 함수
  const [amount, setAmount] = React.useState("0");
  const onChange = (converterValue) => {
    //input의 value를 받아서 amount에 넣는다.
    setAmount(converterValue.target.value);
  }

  // minutes와 hours의 abled를 변환하는 함수
  const [flipped, setFlip] = React.useState(true);
  const onFlip = () => {
    // input에 작성한 값은 0이 된다.
    reset();
    // flipped의 값이 반대로 변환된다.
    // disabled이 true이면 input 사용 못함
    setFlip((current) => !current);
  }

  const reset = () => setAmount("0");
  

  return (
    <div className="conveter">
      <div>
        <label htmlFor="min">Minutes</label>
        <input value={flipped ? amount : amount*60} id="min" placeholder="minutes" type="number" onChange={onChange} disabled={!flipped}></input>
      </div>

      <div>
      <label htmlFor="hours">Hours</label>
      <input value={flipped ? Math.round(amount/60) : amount} id="hours" placeholder="hours" type="number" onChange={onChange} disabled={flipped}></input>
      </div>

      <button onClick={reset}>reset</button>
      <button onClick={onFlip}>flip</button>
    </div>
  )
}
```
